public class SystemQuickSorting {
    public void systemSort(int[] array) {
        long start = System.currentTimeMillis();
        Arrays.sort(array);
        long stop = System.currentTimeMillis();
        long waste = stop - start;
        System.out.println("Time wasted: " + waste / 1000f);
    }
}
